class DNNModel(nn.Module):
    def __init__(self):
        super(DNNModel, self).__init__()
        self.fc1 = nn.Linear(28*28, 128)
        self.fc2 = nn.Linear(128, 32)
        self.output = nn.Linear(32, 10)

    def forward(self, x):
        #     ( )
        x = x.view(-1, 28*28)
        x = F.relu(self.fc1(x))
        x = F.relu(self.fc2(x))
        x = self.output(x)
        return x

def model_train(model, data_loader, loss_fn, optimizer, device):
    model.train()

    # 
    total_batch_loss = 0   #  
    total_num = 0          #  

    prograss_bar = tqdm(data_loader)

    # mini-batch 
    for img, lbl in prograss_bar:
        # image, label  device .
        img, lbl = img.to(device), lbl.to(device)

        optimizer.zero_grad()  #  
        output = model(img)    # Forward Propagation .  .

        loss = loss_fn(output, lbl) #  
        loss.backward()   # (Back Propagation) .   
        optimizer.step()  #  

        # output max(dim=1) max probability max index .
        # max probability , max index pred 
        _, pred = output.max(dim=1)

        # pred.eq(lbl).sum()    label  .
        # item() tensor  .
        total_num += pred.eq(lbl).sum().item() #

        #    Epoch       loss .
        # total_batch_loss 1   loss 
        total_batch_loss += loss.item() * img.size(0)

    acc = total_num / len(data_loader.dataset) #  

    #  (loss)  .
    # train_loss, train_acc
    return total_batch_loss / len(data_loader.dataset), acc

def model_evaluate(model, data_loader, loss_fn, device):
    model.eval()  # model.eval()    

    with torch.no_grad():
        #     
        total_num = 0
        total_batch_loss = 0

        #  evaluation 
        for img, lbl in data_loader:

            # device  
            img, lbl = img.to(device), lbl.to(device)

            output = model(img) # Forward Propagation .  .

            # output max(dim=1) max probability max index .
            _, pred = output.max(dim=1)
            total_num += torch.sum(pred.eq(lbl)).item()  #    ()

            #    Epoch       loss .
            total_batch_loss += loss_fn(output, lbl).item() * img.size(0)

        acc = total_num / len(data_loader.dataset)  #  

        #   - val_loss, val_acc
        return total_batch_loss / len(data_loader.dataset), acc
